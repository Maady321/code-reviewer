// Update user details
exports.updateProfile = async (req, res) => {
    const { name, email } = req.body;

    // Bad performance due to not using bulk update
    try {
        let user = await User.findById(req.user.id);
        if (!user) return res.status(404).json({ msg: 'User not found' });

        user.name = name;
        user.email = email;
        
        await user.save();
        res.json(user);
    } catch (err) {
        console.error(err.message);
        res.status(500).send('Server Error');
    }
}

Performance Optimization

Performing a findById followed immediately by save() makes two database trips. Use `findOneAndUpdate` to reduce latency.
